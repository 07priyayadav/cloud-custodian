.. _aws.launch-config.filters.age:

age
+++
Filter ASG launch configuration by age (in days)

:example:

.. code-block:: yaml

        policies:
          - name: asg-launch-config-old
            resource: launch-config
            filters:
              - type: age
                days: 90
                op: ge

.. container:: toggle

  

  .. raw:: html
     
    <div class="header docutils container" style=""></div>

  .. code-block:: yaml

    properties:
      days:
        type: number
      op:

        - eq
        - equal
        - ne
        - not-equal

        - greater-than

        - gte
        - le
        - lte
        - lt
        - less-than
        - glob

        - regex-case

        - ni
        - not-in
        - contains
        - difference
        - intersect
      type:
        enum:
        - age

    - type


.. _aws.launch-config.filters.unused:

unused
++++++
Filters all launch configurations that are not in use but exist

:example:

.. code-block:: yaml

        policies:
          - name: asg-unused-launch-config
            resource: launch-config
            filters:
              - unused

.. _aws.launch-config.actions.delete:

delete
++++++
Filters all unused launch configurations

:example:

.. code-block:: yaml

        policies:
          - name: asg-unused-launch-config-delete
            resource: launch-config
            filters:
              - unused
            actions:
              - delete
